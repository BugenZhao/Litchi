21  struct FPStruct {
22  char signature[4]; // "_MP_"
23  physaddr_t configuration_table;
24  uint8_t length; // In 16 bytes (e.g. 1 = 16 bytes, 2 = 32 bytes)
25  uint8_t mp_specification_revision;
26  uint8_t checksum; // This value should make all bytes in the table equal 0 when added together
27  uint8_t default_configuration; // If this is not zero then configuration_table should be
28  // ignored and a default configuration should be loaded instead
29  uint32_t features; // If bit 7 is then the IMCR is present and PIC mode is being used, otherwise
30  // virtual wire mode is; all other bits are reserved
31  } __attribute__((__packed__));

78  template<typename T>
79  static bool checksumOk(const T *addr, size_t len = sizeof(T)) {
80  uint8_t sum = 0;
81  for (size_t i = 0; i < len; ++i) {
82  sum += ((uint8_t *) addr)[i];
83  }
84  return sum == 0;
85  }

87  // Search for the Floating Pointer Structure
88  static FPStruct *searchFP() {
89  // Search FPStruct in [addr, addr+len)
90  auto search = [](physaddr_t addr, size_t len) -> FPStruct * {
91  auto end = addr + len;
92  for (; addr < end; addr += sizeof(FPStruct)) {
93  auto fp = static_cast<FPStruct *>(KERN_ADDR(addr));
94  // check signature and checksum
95  if (mem::cmp(fp->signature, "_MP_", 4) == 0 && checksumOk(fp)) {
96  console::out::print(" Found FPStruct at %p\n", fp);
97  return fp;
98  }
99  }
100  return nullptr;
101  };
102 
103  // there are 3 possible positions
104  // 1. Check the first KB of the Extended BIOS Data Area (EBDA)
105  constexpr uint16_t bdaSeg = 0x40;
106  auto bda = (uint8_t *) KERN_ADDR(bdaSeg << 4);
107  if (auto ebdaSeg = *(uint16_t *) (bda + 0x0e); ebdaSeg) { // EBDA exists
108  auto ebdaPhy = ebdaSeg << 4;
109  if (auto fp = search(ebdaPhy, 1024); fp) return fp;
110  }
111 
112  // 2. No EBDA, check the last KB of system base memory
113  auto phy = *(uint16_t *) (bda + 0x13) * 1024;
114  if (auto fp = search(phy - 1024, 1024); fp) return fp;
115 
116  // 3. Check the BIOS read-only memory space between 0xe0000 and 0xfffff
117  return search(0xe0000, 0xfffff - 0xe0000);
118  }

129  void init() {
130  console::out::print("Initializing multi-processor...\n");
131 
132  auto fp = searchFP();
133  if (fp == nullptr) {
134  console::out::print("No multi-processor support.\n");
135  cpuCount = 1;
136  cpus[0].id = 0;
137  cpus[0].status = CPUInfo::Status::started;
138  bootstrapCPU = &cpus[0];
139  return;
140  }
141 
142  // get Configuration Table
143  auto conf = (ConfTable *) KERN_ADDR(fp->configuration_table);
144  assert(conf != nullptr); // no default configuration support
145  assert(mem::cmp(conf->signature, "PCMP", 4) == 0); // signature
146  assert(checksumOk(conf, conf->length)); // base checksum
147  assert(checksumOk((uint8_t *) conf + conf->length, conf->extended_table_length)); // extended checksum
148 
149  // APIC OEM ID
150  char oem[9] = {};
151  mem::copy(oem, conf->oem_id, sizeof(conf->oem_id));
152  console::out::print(" MP Conf Table OEM: %s\n", oem);
153 
154  // save local APIC MMIO physical address
155  lapic::lapicPhy = conf->lapic_address;
156 
157  // **After** the table, there are entry_count entries describing more information about the system
158  uintptr_t entries = (uintptr_t) conf + sizeof(ConfTable);
159  ProcessorAPIC *proc;
160  bool isBootstrap;
161  console::out::print(" Found CPU");
162 
163  for (int i = 0; i < conf->entry_count; ++i) {
164  switch (*(APICType *) entries) {
165  case MPPROC: // processor entry type
166  proc = (ProcessorAPIC *) entries;
167  isBootstrap = proc->flags & 0x02;
168 
169  if (cpuCount < MAX_CPU_COUNT) { // add this CPU
170  cpus[cpuCount].id = cpuCount;
171  console::out::print(" %d%s", proc->local_apic_id, isBootstrap ? "[B]" : "");
172  if (isBootstrap) bootstrapCPU = &cpus[cpuCount];
173  cpuCount++;
174  }
175 
176  entries += sizeof(ProcessorAPIC);
177  break;
178 
179  default: // other apic entry
180  entries += 8;
181  break;
182  }
183  }
184 
185  bootstrapCPU->status = CPUInfo::Status::started;
186  isMP = cpuCount > 1;
187 
188  lapic::init();
189 
190  console::out::print("\n %<Done\n", WHITE);
191  }
192 }

239  void init() {
240  using trap::Type;
241 
242  // no MP support
243  if (!lapicPhy) return;
244 
245  // lapicPhy is the physical address of the LAPIC's 4K MMIO
246  // region. Map it in to virtual memory so we can access it.
247  lapicKern = static_cast<volatile uint32_t *>(vmem::pgdir::mmioMap(lapicPhy, 4096));
248 
249  // Enable local APIC; set spurious interrupt vector.
250  lapicw(SVR, ENABLE | (uint32_t) Type::hwSpurious);
251 
252  // The timer repeatedly counts down at bus frequency
253  // from lapic[TICR] and then issues an interrupt.
254  // If we cared more about precise timekeeping,
255  // TICR would be calibrated using an external time source.
256  lapicw(TDCR, X1);
257  lapicw(TIMER, PERIODIC | (uint32_t) Type::hwTimer);
258  lapicw(TICR, 10000000);
259 
260  // Leave LINT0 of the BSP enabled so that it can get
261  // interrupts from the 8259A chip.
262  //
263  // According to Intel MP Specification, the BIOS should initialize
264  // BSP's local APIC in Virtual Wire Mode, in which 8259A's
265  // INTR is virtually connected to BSP's LINTIN0. In this mode,
266  // we do not need to program the IOAPIC.
267  if (mp::thisCPU() != mp::bootstrapCPU)
268  lapicw(LINT0, MASKED);
269 
270  // Disable NMI (LINT1) on all CPUs
271  lapicw(LINT1, MASKED);
272 
273  // Disable performance counter overflow interrupts
274  // on machines that provide that interrupt entry.
275  if (((lapicKern[VER] >> 16) & 0xFF) >= 4)
276  lapicw(PCINT, MASKED);
277 
278  // Map error interrupt to IRQ_ERROR.
279  lapicw(ERROR, (uint32_t) Type::hwError);
280 
281  // Clear error status register (requires back-to-back writes).
282  lapicw(ESR, 0);
283  lapicw(ESR, 0);
284 
285  // Ack any outstanding interrupts.
286  lapicw(EOI, 0);
287 
288  // Send an Init Level De-Assert to synchronize arbitration ID's.
289  lapicw(ICRHI, 0);
290  lapicw(ICRLO, BCAST | INIT | LEVEL);
291  while (lapicKern[ICRLO] & DELIVS);
292 
293  // Enable interrupts on the APIC (but not on the processor).
294  lapicw(TPR, 0);
295  }

308 // Start additional processor running entry code at addr.
309 // See Appendix B of MultiProcessor Specification.
310  void startAP(uint8_t apicid, uint32_t addr) {
311  using namespace x86;
312 
313  int i;
314  uint16_t *wrv;
315 
316  // "The BSP must initialize CMOS shutdown code to 0AH
317  // and the warm reset vector (DWORD based at 40:67) to point at
318  // the AP startup code prior to the [universal startup algorithm]."
319  outb(IO_RTC, 0xF); // offset 0xF is shutdown code
320  outb(IO_RTC + 1, 0x0A);
321  wrv = (uint16_t *) KERN_ADDR(0x40 << 4 | 0x67); // Warm reset vector
322  wrv[0] = 0;
323  wrv[1] = addr >> 4;
324 
325  // "Universal startup algorithm."
326  // Send INIT (level-triggered) interrupt to reset other CPU.
327  lapicw(ICRHI, apicid << 24);
328  lapicw(ICRLO, INIT | LEVEL | ASSERT);
329  microdelay(200);
330  lapicw(ICRLO, INIT | LEVEL);
331  microdelay(100); // should be 10ms, but too slow in Bochs!
332 
333  // Send startup IPI (twice!) to enter code.
334  // Regular hardware is supposed to only accept a STARTUP
335  // when it is in the halted state due to an INIT. So the second
336  // should be ignored, but it is part of the official Intel algorithm.
337  // Bochs complains about the second one. Too bad for Bochs.
338  for (i = 0; i < 2; i++) {
339  lapicw(ICRHI, apicid << 24);
340  lapicw(ICRLO, STARTUP | (addr >> 12));
341  microdelay(200);
342  }
343  }

345  void lapic_ipi(int vector) {
346  lapicw(ICRLO, OTHERS | FIXED | vector);
347  while (lapicKern[ICRLO] & DELIVS);
348  }
