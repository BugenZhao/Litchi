49  static void init(void) {
50  // Turn off the FIFO
51  x86::outb(COM1 + COM_FCR, 0);
52 
53  // Set speed; requires DLAB latch
54  x86::outb(COM1 + COM_LCR, COM_LCR_DLAB);
55  x86::outb(COM1 + COM_DLL, (uint8_t) (115200 / 9600));
56  x86::outb(COM1 + COM_DLM, 0);
57 
58  // 8 data bits, 1 stop bit, parity off; turn off DLAB latch
59  x86::outb(COM1 + COM_LCR, COM_LCR_WLEN8 & ~COM_LCR_DLAB);
60 
61  // No modem controls
62  x86::outb(COM1 + COM_MCR, 0);
63  // Enable rcv interrupts

155  // Put char into cga. Explicit colors will be used only if no color info in c
156  static void putChar(int c, enum color_t defForeColor, enum color_t defBackColor) {
157  if ((c & 0xff00) == 0) // no color in c
158  c = COLOR_CHAR(c, defForeColor, defBackColor);
159 
160  // Print the character
161  switch (c & 0xff) {
162  case '\b':
163  if (pos > 0) buffer[--pos] = (c & ~0xff) | ' ';
164  break;
165  case '\t':
166  putChar(' ', defForeColor, defBackColor);
167  putChar(' ', defForeColor, defBackColor);
168  putChar(' ', defForeColor, defBackColor);
169  putChar(' ', defForeColor, defBackColor);
170  break;
171  case '\n':
172  pos += CRT_COLS;
173  // LF only
174  case '\r':
175  pos = pos / CRT_COLS * CRT_COLS;
176  break;
177  case '\0':
178  break;
179  default:
180  buffer[pos++] = c;
181  }
182 
183  // Scroll the screen
184  if (pos == CRT_SIZE) {
185  for (int i = 0; i < CRT_ROWS - 1; ++i) {
186  mem::copy(buffer + i * CRT_COLS, buffer + (i + 1) * CRT_COLS, CRT_COLS * sizeof(uint16_t));
187  }
188  for (int j = CRT_SIZE - CRT_COLS; j < CRT_SIZE; ++j) {
189  buffer[j] = (c & ~0xff) | ' ';
190  }
191  pos -= CRT_COLS;
192  }
193 
194  // Move the cursor
195  cursorRefresh();
196  }
