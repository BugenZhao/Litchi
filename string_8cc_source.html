28  void *move(void *dest, const void *src, size_t count) {
29  // https://www.student.cs.uwaterloo.ca/~cs350/common/os161-src-html/doxygen/html/memmove_8c_source.html
30  size_t i;
31  /*
32  * If the buffers don't overlap, it doesn't matter what direction
33  * we copy in. If they do, it does, so just assume they always do.
34  * We don't concern ourselves with the possibility that the region
35  * to copy might roll over across the top of memory, because it's
36  * not going to happen.
37  *
38  * If the destination is above the source, we have to copy
39  * back to front to avoid overwriting the data we want to
40  * copy.
41  *
42  * dest: dddddddd
43  * src: ssssssss ^
44  * | ^ |___|
45  * |___|
46  *
47  * If the destination is below the source, we have to copy
48  * front to back.
49  *
50  * dest: dddddddd
51  * src: ^ ssssssss
52  * |___| ^ |
53  * |___|
54  */
55  if ((uintptr_t) dest < (uintptr_t) src) {
56  /*
57  * As author/maintainer of libc, take advantage of the
58  * fact that we know memcpy copies forwards.
59  */
60  return mem::copy(dest, src, count);
61  }
62  /*
63  * Copy by words in the common case. Look in memcpy.c for more
64  * information.
65  */
66  if ((uintptr_t) dest % sizeof(long) == 0 &&
67  (uintptr_t) src % sizeof(long) == 0 &&
68  count % sizeof(long) == 0) {
69  long *d = static_cast<long *>(dest);
70  const long *s = static_cast<const long *>(src);
71  /*
72  * The reason we copy index i-1 and test i>0 is that
73  * i is unsigned -- so testing i>=0 doesn't work.
74  */
75  for (i = count / sizeof(long); i > 0; i--) {
76  d[i - 1] = s[i - 1];
77  }
78  } else {
79  char *d = static_cast<char *>(dest);
80  const char *s = static_cast<const char *>(src);
81  for (i = count; i > 0; i--) {
82  d[i - 1] = s[i - 1];
83  }
84  }
85  return dest;
86  }

151  // Note: str will be split up to (bufCount - 1) pieces,
152  // since the last char * will always be NULL
153  int split(char *str, const char *delimiters, char **resultBuf, size_t bufCount, bool ignoreQuo) {
154  if (str == NULL) return -1;
155  size_t count = 0;
156  int quo = 0;
157 
158  while (count < bufCount - 1) {
159  while (*str && find(delimiters, *str) != NULL) *str++ = '\0';
160  if (*str == '\0') break;
161  resultBuf[count++] = str;
162  while (*str && (quo || find(delimiters, *str) == NULL)) {
163  if (!ignoreQuo && *str == '\"') {
164  if (!quo) resultBuf[count - 1] = str + 1;
165  quo = !quo;
166  *str = '\0';
167  }
168  str++;
169  }
170  }
171 
172  if (quo) return -1;
173 
174  resultBuf[count] = NULL;
175  return count;
176  }
