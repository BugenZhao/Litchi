20  // Interrupt DESCRIPTOR Table <= generate from ivt
21  GateDesc idt[256] = {{0}};

27  // Task State, if trapped, CPU will automatically load ss, esp from it
28  // we need to set them to a new, real "kernel stack" (instead of current bootstrap stack),
29  // then describe it in the last segment descriptor(GD_TSS0) in GDT,
30  // finally, load the descriptor number to "Task Register"
31  TaskState kernelTS;

33  void init() {
34  print("Initializing traps...\n");
35 
36  for (int i = 0; i < 256; ++i) {
37  // gate : interrupt descriptor
38  // istrap: trap or fault (different behaviors on restart), see IA32 manual p5-6 [ignored]
39  // sel : GD_KT, irq's cs
40  // offset: ivt[i], irq's ip
41  // dpl : privilege, all traps but our system call should not be directly called by user
42  SETGATE(idt[i], 0, GD_KT, ivt[i], 0);
43  }
44 
45  // allow some traps be called by user => dpl = 3
46  for (auto no: {(int) Type::syscall,
47  (int) Type::breakpoint,
48  (int) Type::debug}) {
49  SETGATE(idt[no], 1, GD_KT, ivt[no], 3);
50  }
51 
52  // set and load tss, ivt per CPU
53  initPerCpu();
54 
55  print(" %<Done\n", WHITE);
56  }

58  void initPerCpu() {
59  // see comments around the definition of kernelTS
60  // Note: for time-slicing multitasking, the stack(esp0) should be switched every context switching
61  kernelTS.ts_ss0 = GD_KD;
62  kernelTS.ts_esp0 = KSTACKTOP;
63  kernelTS.ts_iomb = sizeof(TaskState); // just ignored, http://forum.osdev.org/viewtopic.php?t=13678
64 
65  // specifying the TSS' address as "base",
66  // TSS' size as "limit",
67  // 0x89 (Present|Executable|Accessed) as "access byte"
68  // and 0x40 (Size-bit) as "flags". In the TSS itself
69  // https://wiki.osdev.org/TSS
70  gdt[GD_TSS0 >> 3] = SEG16(STS_T32A, (uint32_t) &kernelTS, sizeof(TaskState) - 1, 0);
71  gdt[GD_TSS0 >> 3].sd_s = 0;
72 
73  // load
74  x86::ltr(GD_TSS0);
75  x86::lidt(&idtPD);
76  }
77 }

80  [[noreturn]] void trap(Frame *tf) {
81  using namespace task;
82 
83  // interrupts should be disabled
84  assert(!(x86::read_eflags() & FL_IF));
85 
86  if ((tf->cs & 0b11) == 3) { // trapped from user mode, must be Task::current
87  Task::current->trapFrame = *tf; // update tf
88  tf = &Task::current->trapFrame; // use the one on kernel stack to avoid problems
89 
90  if (tf->trapType != Type::syscall)
91  print("[%08x] Back to kernel: Trap [%s]\n", Task::current->id, describe(tf->trapType));
92  } else {
93  print("Kernel Trap [%s]\n", describe(tf->trapType));
94  }
95 
96  // dispatch to trap handlers
97  tf->dispatch();
98 
99  // run again
100  assert(Task::current);
101  Task::current->run(tf->trapType != Type::syscall);
102  }
103 }

118  break;
119  case Type::syscall:

122 
123  case Type::debug:
124  case Type::breakpoint:

126  break;

128  case Type::divide:
129  case Type::nmi:
130  case Type::overflow:
131  case Type::bound:
132  case Type::invalidOp:
133  case Type::device:

144  default:

160  // get fault virtual address
161  auto faultVa = x86::rcr2();
162 
163  if ((tf->cs & 0b11) == 0) {
164  kernelPanic("Unhandled page fault (%p) from kernel\n", faultVa);
165  } else {
166  print("[%08x] Unhandled page fault (at %p) [%s, %s, %s]\n",
167  task::Task::current->id, faultVa,
168  tf->err & 4 ? "user" : "kernel",
169  tf->err & 2 ? "write" : "read",
170  tf->err & 1 ? "protection" : "not-present");
171  task::Task::current->destroy(true); // will trap into monitor
172  }
173  }
