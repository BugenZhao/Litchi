225  Result parseCmd(char *cmd, trap::Frame *tf) {
226  char *argv[MAX_ARGS + 2];
227  int argc = str::splitWs(cmd, argv, MAX_ARGS + 2);
228 
229  if (argc == MAX_ARGS + 1) return Result::tooManyArgs;
230  else if (argc == -1) return Result::badSyntax;
231 
232  for (size_t i = 0; i < ARRAY_SIZE(commands); ++i) {
233  if (str::cmpCase(argv[0], commands[i].cmd) == 0) {
234  return commands[i].func(argc, argv, tf);
235  }
236  }
237  return Result::badCommand;
238  }
