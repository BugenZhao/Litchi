137  // Initialize pageInfoList
138  void init() {
139  // Create page info list
140  PageInfo::array = (struct PageInfo *) bootAlloc(sizeof(struct PageInfo) * nPages);
141  mem::clear(PageInfo::array, sizeof(struct PageInfo) * nPages);
142 
143  // 1. Mark all pages below 1MB as used
144  size_t i;
145  for (i = 0; i < EXTPHYSMEM / PGSIZE; ++i)
146  PageInfo::array[i].refCount = 1;
147 
148  // 2. Mark pages where kernel resides as used
149  size_t nextFreePage = (physaddr_t) PHY_ADDR(bootAlloc(0)) / PGSIZE;
150  for (; i < nextFreePage; ++i)
151  PageInfo::array[i].refCount = 1;
152 
153  size_t j;
154  // 3. Mark [4MB, total] as free
155  for (j = 4 * 1024 * 1024 / PGSIZE; j < nPages; ++j) {
156  PageInfo::array[j].refCount = 0;
157  PageInfo::array[j].nextFree = PageInfo::freeList;
158  PageInfo::freeList = PageInfo::array + j;
159  }
160 
161  // CRITICAL:
162  // 4. Mark [nextFreePage * PGSIZE, 4MB) as free **AT LAST**,
163  // since we cannot touch higher memory now when we are using entry_pgdir
164  for (; i < 4 * 1024 * 1024 / PGSIZE; ++i) {
165  PageInfo::array[i].refCount = 0;
166  PageInfo::array[i].nextFree = PageInfo::freeList;
167  PageInfo::freeList = PageInfo::array + i;
168  }
169 
170  // Do some test
171  assert(PageInfo::fromPhy(0xb8000)->refCount > 0);
172  assert(PageInfo::fromPhy(PHY_ADDR(PageInfo::array))->refCount > 0);
173  assert(PageInfo::fromPhy((totalMem - 4) * 1024u)->refCount == 0);
174  // Make sure we are about to alloc page below 4MB !!!
175  assert(PDX(PageInfo::freeList->toPhy()) == 0);

194  // Find the page table entry of va, but won't do any mapping
195  // If not exist but create == true, func will create one page table / page dir entry
196  pte_t *findPte(pde_t *pageDir, const void *va, bool create) {
197  uint32_t pdx = PDX(va); // Page directory index
198  uint32_t ptx = PTX(va); // Page table index
199  pde_t *pde = pageDir + pdx; // Page directory entry
200 
201  if ((*pde) & PTE_P) {
202  // Entry present
203  // Clear the least significant 10 bits (flags) to get the pt address,
204  // then convert it to virtual address in order to access
205  pte_t *pageTable = (pte_t *) KERN_ADDR(PTE_ADDR(*pde));
206  return pageTable + ptx;
207  } else if (!create) {
208  return nullptr;
209  } else {
210  // Allocate a new **PAGE TABLE** page's info
211  struct PageInfo *pp = PageInfo::alloc(true);
212  if (pp == nullptr) { return nullptr; } // Allocation fails
213  pp->refCount++;
214 
215  mem::clear(pp->toKernV(), PGSIZE); // Clear the real page
216 
217  // Actually insert the page table page into pageDir+idx (*pde)
218  *pde = pp->toPhy() | PTE_P | PTE_U | PTE_W; // Set permissive flags
219 
220  invalidateTLB(pageDir, (void *) va);
221 
222  pte_t *pageTable = (pte_t *) KERN_ADDR(PTE_ADDR(*pde));
223  return pageTable + ptx;
224  }
225  }

227  // Lookup physical page mapped at va
228  std::tuple<PageInfo *, pte_t *> findInfo(pde_t *pageDir, const void *va) {
229  pte_t *pte = findPte(pageDir, va, 0);
230  // No page or page not present
231  if (pte == nullptr || !((*pte) & PTE_P)) return {nullptr, nullptr};
232  // Get pa from pte, then get page info from pa
233  return {PageInfo::fromPhy(PTE_ADDR(*pte)), pte };
234  }

236  // Remove physical page mapped at va
237  void remove(pde_t *pageDir, void *va) {
238  auto [pp, pte] = findInfo(pageDir, va);
239  if (pp && (*pte) & PTE_P) {
240  // Dec the ref, and mark the physical page free if necessary
241  pp->decRef();
242  // Clear page table entry
243  *pte = 0;
244  // Invalidate TLB since we removed an entry
245  invalidateTLB(pageDir, va);
246  }
247  }

249  // Map the physical page 'pp' at va, with permission bits 'perm'
250  Result insert(pde_t *pageDir, struct PageInfo *pp, void *va, int perm) {
251  pte_t *pte = findPte(pageDir, va, 1);
252  if (pte == nullptr) return Result::fuck;
253 
254  // First, inc refCount of pp
255  pp->refCount++;
256  if ((*pte) & PTE_P) {
257  // Entry already present
258  // Corner-case won't cause bug since we've already inc the ref count,
259  // then it (pp) won't be freed.
260  remove(pageDir, va);
261  }
262  *pte = pp->toPhy() | perm | PTE_P;
263  // Invalidate TLB since we may have updated an entry
264  invalidateTLB(pageDir, va);
265  return Result::ok;
266  }

268  // Statically map [va, va+size) to [pa, pa+size)
269  // It will not make changes on PageInfo::array
270  void staticMap(pde_t *pageDir, uintptr_t va, size_t size, physaddr_t pa, int perm) {
271  uint32_t offset = 0;
272  // There might be multiple pte's to find and set
273  while (offset < size) {
274  // Get the page table entry
275  pte_t *pte = findPte(pageDir, (const void *) (va + offset), 1);
276  // Actually set the pte to map the page
277  *pte = (pa + offset) | PTE_P | perm;
278  invalidateTLB(pageDir, (void *) (va + offset));
279  offset += PGSIZE;
280  }
281  }

283  // Statically map [pa, pa+size) to some address in [MMIOBASE, MMIOLIM)
284  void *mmioMap(physaddr_t pa, size_t size) {
285  static uintptr_t next = MMIOBASE;
286 
287  size = ROUNDUP(size, PGSIZE);
288  if (next + size >= MMIOLIM || next + size < MMIOBASE)
289  kernelPanic("Cannot map vm for mmio (pa %p, size %u)", pa, size);
290 
291  // cache-disable, write-through
292  staticMap(kernelPageDir, next, size, pa, PTE_W | PTE_PCD | PTE_PWT);
293 
294  next += size;
295  return (void *) (next - size);

298  static void init() {
299  // Map PageInfo::array at va UPAGES for user (read-only)
300  staticMap(kernelPageDir, UPAGES, PTSIZE, PHY_ADDR(PageInfo::array), PTE_U);
301  // Map kernel stack at va KSTACKTOP-KSTKSIZE
302  staticMap(kernelPageDir, KSTACKTOP - KSTKSIZE, KSTKSIZE, PHY_ADDR(bootstack), PTE_W);
303  // Map all of physical memory in [0, 2^32 - KERNBASE) at va KERNBASE
304  staticMap(kernelPageDir, KERNBASE, ~KERNBASE, 0, PTE_W);
305 
306  // Actually load page dir

308 
309  // Set other cr0 flags

312  cr0 &= ~(CR0_TS | CR0_EM);

314 
315  // Do some checks
316  assert(*(uint64_t *) UPAGES == *(uint64_t *) PageInfo::array);
317  assert(*(uint32_t *) (KSTACKTOP - (bootstacktop - (char *) &cr0)) == cr0);

350  // some user task may trick the kernel, using syscall to access kernel space
351  // use this to check if [va, va + size) all satisfy the perm
352  bool userCheck(pde_t *pageDir, const void *va, size_t size, int perm) {
353  perm = perm | PTE_P;
354  auto va_it = (uint32_t) va;
355 
356  for (; va_it < (uint32_t) va + size; va_it += PGSIZE) {
357  // out of user limit
358  if ((uintptr_t) va_it >= ULIM) goto bad;
359  // perm not satisfied
360  pte_t *pte = findPte(pageDir, (const void *) va_it, false);
361  if (pte == NULL) goto bad;
362  if (!(*pte & perm)) goto bad;
363  }
364  return true;
365  bad:
366  return false;
367  }
