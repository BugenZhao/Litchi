50  // pop the trap frame and goto user mode
51  [[noreturn]] inline void pop() {
52  asm volatile (
53  "movl %0, %%esp\n" // move tf to stack
54  "popal\n"
55  "popl %%es\n" // must be LONG
56  "popl %%ds\n"
57  "addl $0x8, %%esp\n" // skip trap number and err
58  "iret\n" // IRET!!!
59  :
60  :"g"(this) // put this to any General reg
61  :"memory" // do not reorder
62  );
63 
64  kernelPanic("failed to pop and iret trap frame %p", this);
65  }
