51  void initPerCpu() {
52  // now switch gdt from bootstrap gdt to new one, for use of privilege level
53  // see kernel/gdt.c
54  x86::lgdt(&gdtPD);
55 
56  // set segment register (selector) to the correct descriptor
57 
58  // the kernel never uses GS or FS, so we leave those set to the user data segment.
59  // LS 2 bit: Requested Privilege Level
60  asm volatile("movw %%ax,%%gs" : : "a" (GD_UD | 3));
61  asm volatile("movw %%ax,%%fs" : : "a" (GD_UD | 3));
62  // the kernel does use ES, DS, and SS.
63  asm volatile("movw %%ax,%%es" : : "a" (GD_KD));
64  asm volatile("movw %%ax,%%ds" : : "a" (GD_KD));
65  asm volatile("movw %%ax,%%ss" : : "a" (GD_KD));
66 
67  // perform a long jump to set new CS selector
68  asm volatile("ljmp %0,$1f\n 1:\n" : : "i" (GD_KT));
69  }
70 }

73  std::tuple<Task *, Result> Task::alloc(taskid_t parentId) {
74  // find a free slot
75  Task *task = freeList;
76  if (task == nullptr)
77  return {nullptr, Result::noFreeTask};
78  freeList = task->nextFree;
79 
80  // allocate page dir
81  if (Result r; (r = task->setupMemory()) != Result::ok)
82  return {nullptr, r};
83 
84  // allocate a random id, ending up with the #
85  uint32_t rand = lrandom::rand() << 16 | lrandom::rand();
86  task->id = (rand & ~(maxCount - 1)) | (task - array);
87 
88  // set other fields
89  task->parentId = parentId;
90  task->status = TaskStatus::ready;
91  task->type = TaskType::user;
92 
93  // clear the trap frame
94  task->trapFrame.clear();
95  // set segment descriptors to users', 3: RPL, see: initPerCpu()
96  task->trapFrame.ds = GD_UD | 3;
97  task->trapFrame.es = GD_UD | 3;
98  task->trapFrame.cs = GD_UT | 3;
99  task->trapFrame.ss = GD_UD | 3;
100  // set user stack
101  task->trapFrame.esp = USTACKTOP; // user stack top
102 
103  // eip will be set after binary loaded
104 
105  print("[%08x] Allocated task\n", task->id);
106  return {task, Result::ok};
107  }

109  Result Task::setupMemory() {
110  // allocate a page for the task's page dir
111  auto page = vmem::PageInfo::alloc(true);
112  if (page == nullptr) return Result::noMemory;
113 
114  // simply duplicate the kernel's
115  this->pageDir = static_cast<pde_t *>(page->toKernV());
116  mem::copy(page->toKernV(), vmem::kernelPageDir, PGSIZE);
117  page->refCount += 1;
118 
119  // again, make a recursive map to the page dir itself
120  this->pageDir[PDX(UVPT)] = PHY_ADDR(this->pageDir) | PTE_U | PTE_P;
121 
122  return Result::ok;
123  }

125  void Task::free() {
126  if (this == current) x86::lcr3(PHY_ADDR(vmem::kernelPageDir));
127 
128  // free all pages, from 0x0 to UTOP
129  for (int pdx = 0; pdx < (int) PDX(UTOP); ++pdx) {
130  auto pde = this->pageDir[pdx]; // find the page dir entry
131  if (!(pde & PTE_P)) continue; // page table not present
132 
133  auto pa = PTE_ADDR(pde);
134  auto pageTable = static_cast<pte_t *>(KERN_ADDR(pa));
135  for (int ptx = 0; ptx < NPTENTRIES; ++ptx) {
136  auto pte = pageTable[ptx]; // find the page table entry
137  if (pte & PTE_P) // remove the page from page dir, and free the info if needed
138  vmem::pgdir::remove(this->pageDir, PGADDR(pdx, ptx, 0));
139  }
140 
141  // free the page table itself
142  pageDir[pdx] = 0;
143  vmem::PageInfo::fromPhy(pa)->decRef();
144  }
145 
146  // free the page dir itself
147  vmem::PageInfo::fromPhy(PHY_ADDR(pageDir))->decRef();
148 
149  // free the Task
150  status = TaskStatus::free;
151  nextFree = freeList;
152  freeList = this;
153 
154  print("[%08x] Freed task\n", this->id);
155  }

157  void Task::loadElf(uint8_t *binary) {
158  auto elf = (Elf *) binary;
159  if (elf->e_magic != ELF_MAGIC)
160  kernelPanic("bad elf binary");
161 
162  x86::lcr3(PHY_ADDR(this->pageDir));
163 
164  // copy the binary
165  auto ph = (Proghdr *) (binary + elf->e_phoff); // find the elf header
166  auto eph = ph + elf->e_phnum; // end of the header
167  for (; ph < eph; ++ph) {
168  if (ph->p_type != ELF_PROG_LOAD || ph->p_memsz == 0) continue; // not to load
169 
170  // allocate pages
171  this->regionAlloc((void *) ph->p_va, ph->p_memsz);
172  // copy text/data, only copy filesz length
173  auto va = binary + ph->p_offset;
174 // print("VA %p\n", va);
175  mem::copy((void *) ph->p_va, va, ph->p_filesz);
176  }
177 
178  x86::lcr3(PHY_ADDR(vmem::kernelPageDir));
179 
180  // allocate stack (PGSIZE)
181  this->regionAlloc((void *) (USTACKTOP - PGSIZE), PGSIZE);
182  // set eip to elf entry point
183  this->trapFrame.eip = elf->e_entry;
184 
185  print("[%08x] Loaded elf\n", this->id);
186  }

188  // allocate zero-ed pages from _va to _va + len, into this->pageDir
189  void Task::regionAlloc(void *_va, size_t len) {
190  char *va = static_cast<char *>(_va);
191  auto vaLo = ROUNDDOWN(va, PGSIZE);
192  auto vaHi = ROUNDUP(va + len, PGSIZE);
193  if (vaLo >= vaHi) kernelPanic("bad va");
194 
195  // allocate pages
196  for (auto vaIt = vaLo; vaIt < vaHi; vaIt += PGSIZE) {
197  auto page = vmem::PageInfo::alloc(true);
198  if (page == nullptr) kernelPanic("out of memory");
199 // print("PA %p\n", page->toPhy());
200  auto result = vmem::pgdir::insert(this->pageDir, page, vaIt, PTE_U | PTE_W);
201  if (result != Result::ok) kernelPanic("insert page failed");
202  }
203  }
