198 // Read status and data from keyboard register
199 static int _kbdProcessData(void) {
200  // Adopted from JOS/xv6
201  int c;
202  uint8_t stat, data;
203  static uint32_t shift;
204 
205  // Read keyboard status
206  stat = inb(KBSTATP);
207  // Read data in keyboard buffer
208  if ((stat & KBS_DIB) == 0)
209  return -1;
210  // Ignore data from mouse.
211  if (stat & KBS_TERR)
212  return -1;
213 
214  // Read keyboard data
215  data = inb(KBDATAP);
216 
217  if (data == 0xE0) {
218  // E0 escape character
219  shift |= E0ESC;
220  return 0;
221  } else if (data & 0x80) {
222  // Key released
223  data = (shift & E0ESC ? data : data & 0x7F);
224  shift &= ~(shiftcode[data] | E0ESC);
225  return 0;
226  } else if (shift & E0ESC) {
227  // Last character was an E0 escape; or with 0x80
228  data |= 0x80;
229  shift &= ~E0ESC;
230  }
231 
232  shift |= shiftcode[data];
233  shift ^= togglecode[data];
234 
235  c = charcode[shift & (CTL | SHIFT)][data];
236  if (shift & CAPSLOCK) {
237  if ('a' <= c && c <= 'z')
238  c += 'A' - 'a';
239  else if ('A' <= c && c <= 'Z')
240  c += 'a' - 'A';
241  }
242 
243  // Process special keys
244  // Ctrl-Alt-Del: reboot
245  if (!(~shift & (CTL | ALT)) && c == KEY_DEL) {
246  consoleErrPrintC("Rebooting!\n");
247  outb(0x92, 0x3); // courtesy of Chris Frost
248  }
249 
250  return c;
251 }
